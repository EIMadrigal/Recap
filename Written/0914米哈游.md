```cpp
#include <bits/stdc++.h>
using namespace std;
int main(int argc, char* argv[]) {
    int n,k;
    cin>>n>>k;
    string s;
    cin>>s;
    if (s.length() < 12) {
        cout << -1<<"\n";
        return 0;
    }
    vector<int> ids;
    int i = 0;
    while (i<s.length()) {
        int p = s.find("mihoyo", i);
        if(p!=string::npos) {
            ids.push_back(p);
            i = p + 1;
        } else break;
    }
    if(ids.size() < k) {
        cout << -1<<"\n";
        return 0;
    }
    int res=INT_MAX,resi=0,resj=0;
    for (int j=0;j+k-1<ids.size();j++) {
        if (ids[j+k-1]-ids[j]<res){
            resi=ids[j], resj=ids[j+k-1]+5;
            res=ids[j+k-1]-ids[j];
        }
    }
    cout << resi<<" " <<resj<<"\n";
    return 0;
}
```

```cpp
int main(int argc, char* argv[]) {

    cin>>n;
    vector<int> a(n);
    for(int i=0;i<n;++i) {
        cin>>a[i];
    }
    cin>>x>>y;
    sort(a.begin(),a.end());
    if(x==0) {
        cout<<"infinity";
        return 0;
    }
    int l = a[y-1],r=a[n-x];
    if(r<=l) cout<<"0";
    else cout<<r-l;
    return 0;
}
```
