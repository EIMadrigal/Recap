## 订书
```cpp
#include <bits/stdc++.h>
using namespace std;

int main(int argc, char* argv[]) {
    int n,m,tmp;
    cin>>n>>m;
    set<int> s;
    for (int i = 0;i < m;++i) {
        cin >> tmp;
        s.insert(tmp);
    }
    for (auto it = s.begin(); it != s.end(); ) {
        int start = *it, cur = *it;
        it++;
        while (it != s.end() && *it == cur + 1) {
            cur = *it;
            ++it;
        }
        if (start != cur) cout << start << "-" << cur;
        else cout << start;
        if (it != s.end()) cout << ",";
    }
    return 0;
}
```
