## 编程
任务(开始时间, 结束时间, 价值)调度，求选哪些任务可以达到最大的价值，不能区间重叠，返回选中任务下标。按照单位价值排序，然后不断扩展已有的左右边界，57%，剩10分钟了发现这是个假算法。。。大佬们说是LC 1235，看了下是个Hard，失敬了。。。但是DP好像没法知道具体的序列，下面做法具体序列是错的，不会做。。。
```cpp
#include <iostream>
#include <vector>
#include <numeric>
#include <limits>
#include <algorithm>

using namespace std;

class Solution {
public:
    vector<int> schedule(vector<vector<int>> arr) {
        if (arr.empty()) {
          return {};
        }
        int n = arr.size();
        vector<vector<int>> nums(arr);
        for (int i = 0; i < n; ++i) {
            nums[i].push_back(i);
        }
        // 按结束时间排序
        sort(nums.begin(), nums.end(), [](const vector<int>& a, const vector<int>& b) {
            return a[1] < b[1];
        });
        vector<int> res;
        vector<int> dp(n + 1, 0);
        vector<int> prev(n + 1, -1);
        for (int i = 1; i <= n; ++i) {
            int l = 1, r = n + 1;
            while (l < r) {
                int m = l + (r - l) / 2;
                if (nums[i - 1][0] >= nums[m - 1][1]) {
                    l = m + 1;
                } else {
                    r = m;
                }
            }
            prev[i] = l - 1;
            dp[i] = max(dp[i - 1], dp[l - 1] + nums[i - 1][2]);
        }
        int cur = n;
        while (cur > 0) {
            res.push_back(nums[cur - 1][3]);
            cur = prev[cur];
        }
        sort(res.begin(), res.end());
        return res;
    }
};

int main() {
    vector<int> res;
    int arr_rows = 0;
    int arr_cols = 0;
    cin >> arr_rows >> arr_cols;
    vector<vector<int>> arr(arr_rows);
    for (int arr_i = 0; arr_i < arr_rows; arr_i++) {
        for (int arr_j = 0; arr_j < arr_cols; arr_j++) {
            int arr_tmp;
            cin >> arr_tmp;
            arr[arr_i].push_back(arr_tmp);
        }
    }
    Solution *s = new Solution();
    res = s->schedule(arr);
    for (int res_i = 0; res_i < res.size(); res_i++) {
        cout << res[res_i] << endl;;
    }
    return 0;
}
```
