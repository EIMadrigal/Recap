运费是累加的
```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> a(n), b(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i] >> b[i];
    }
    int ans = 0;
    int totalMin = INT_MAX;
    
    for (int i = 0; i < n; ++i) {
        if (a[i] < totalMin) {
            ans += a[i];
            totalMin = min(a[i], totalMin) + b[i];
        }
        else {
            ans += totalMin;
            totalMin = min(a[i], totalMin) + b[i];
        }
    }
    cout << ans << "\n";
    return 0;
}
```
