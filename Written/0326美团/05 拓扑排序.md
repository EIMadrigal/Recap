给一个数组l，表示完成每个任务需要的时间，再给出每个任务的前置任务个数及id，求每个任务的最早完成时间。

```
输入  
3       
5 0
10 1 0
8 2 0 1
输出
5 15 23
```
没来得及做...

```cpp
#include <bits/stdc++.h>
using namespace std;

void topo(vector<int>& l, vector<vector<int>>& g, vector<int>& cnt, vector<int>& finish) {
    
    for (int i = 0; i < g.size(); ++i) {
        for (auto j : g[i]) {
            cout << j << " ";
        }
        cout << "\n";
    }
    
    queue<int> q;
    int n = g.size();

    for (int i = 0; i < n; ++i) {
        if (!cnt[i]) {
            q.push(i);
            finish[i] = l[i];
        }
    }
    while (!q.empty()) {
        int cur = q.front(); q.pop();
        for (int i = 0; i < g[cur].size(); ++i) {
            int to = g[cur][i];
            cnt[to]--;
            // 在能够更新finish[to]的前置点中选择最晚的
            finish[to] = max(finish[to], finish[cur]);
            if (cnt[to] == 0) {
                q.push(to);
                finish[to] += l[to];
            }
        }
    }
}

int main() {
    int n, a, b;
    cin >> n;
    // 拓扑排序需要图g和入度cnt
    vector<int> l(n), cnt(n);
    vector<vector<int>> g(n);
    for (int i = 0; i < n; ++i) {
        cin >> l[i];
        cin >> a;
        cnt[i] = a;
        for (int j = 0; j < a; ++j) {
            cin >> b;
            g[b].push_back(i);
        }
    }
    vector<int> finish(n, 0);
    topo(l, g, cnt, finish);
    for (int i = 0; i < n; ++i) {
        cout << finish[i] << " ";
    }
    return 0;
}
```
