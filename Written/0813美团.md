## 模拟 100
```cpp
#include <bits/stdc++.h>
using namespace std;

int main(int argc, char* argv[]) {
    int n, m, k;
    string s;
    cin >> n >> m >> k;
    cin >> s;
    vector<vector<bool>> vis(n, vector<bool>(m, false));
    int p = 0, q = 0, cnt = 1;
    int i = 0;
    vis[0][0] = true;
    for (i = 0; i < k; ++i) {
        if (s[i] == 'W') {
            --p;
        } else if (s[i] == 'A') {
            --q;
        } else if (s[i] == 'S') {
            ++p;
        } else if (s[i] == 'D') {
            ++q;
        }
        if (!vis[p][q]) {
            vis[p][q] = true;
            ++cnt;
        }
        if (cnt == n * m) {
            break;
        }
    }
    if (cnt == n * m) {
        cout << "Yes\n" << i << "\n";
    } else {
        cout << "No\n" << n * m - cnt << "\n";
    }
    return 0;
}
```
