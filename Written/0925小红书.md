using namespace std;

int main() {
    int n,tmp;
    cin>>n;
    set<int> a;
    int res=0;
    for(int i=0;i<n;++i) {
        cin>>tmp;
        a.insert(tmp);
    }
    int last = 0;
    for(auto aa:a) {
        if(aa!=last+1) break;
        res=aa;
        last=aa;
    }
    cout<<res;
    return 0;
}

3. 水题
```cpp
#include <bits/stdc++.h>
using namespace std;

bool solve(string a,string b) {
    int i=0,j=0;
    while (i<a.length() && j < b.length()) {
        if(a[i]==b[j]) {
            ++i,++j;
        } else {
            ++i;
        }
        if (j==b.length()) return true;
    }
    return j==b.length();
}

int main() {
    int n;
    cin>>n;
    string a,b;
    for(int i=0;i<n;++i) {
        cin>>a>>b;
        if(solve(a,b)) cout<<"yes\n";
        else cout<<"no\n";
    }
    return 0;
}
```
