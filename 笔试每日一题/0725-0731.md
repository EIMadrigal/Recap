## 0725 IP地址与int整数的转换
```cpp
#include <bits/stdc++.h>
using namespace std;

// 10.0.3.193 -> 167773121
// 将10左移24位, 0左移16位, 3左移8位, 193左移0位. 4段按位或即可
unsigned int ipToInt(string ip) {
    const int len = ip.length();
    vector<int> ipList;
    for (int i = 0; i < len; ++i) {
        int j = i;
        while (j < len && ip[j] != '.') {
            j++;
        }
        ipList.push_back(stoi(ip.substr(i, j - i)));
        i = j;
    }
    unsigned res = 0;
    for (int i = 0; i < ipList.size(); ++i) {
        res = res << 8 | ipList[i];
    }
    return res;
}

// 167773121 -> 10.0.3.193
// 与255按位与得193, 右移8位与255按位与得3, 右移8位按位与得0, 右移8位按位与得10
string intToIp(unsigned num) {
    vector<string> ipList;
    for (int i = 0; i < 4; ++i) {
        string seg = to_string(num & 255);
        ipList.push_back(seg);
        num >>= 8;
    }
    string res = "";
    reverse(ipList.begin(), ipList.end());
    for (int i = 0; i < 4; ++i) {
        res += ipList[i];
        if (i < 3) {
            res += '.';
        }
    }
    return res;
}

int main(int argc, char* argv[]) {
    string ip;
    unsigned int num;
    cin >> ip;
    cin >> num;
    cout << ipToInt(ip) << "\n";
    cout << intToIp(num) << "\n";
    return 0;
}
```

## 0726 字符串相减（大数相减）
给定两个字符串形式的非负整数 num1 和num2 ，计算它们的差。num1和num2都只会包含数字0-9，num1和num2都不包含任何前导零，你不能使用任何内建 BigInteger库
```cpp
#include <bits/stdc++.h>
using namespace std;

bool isLess(string num1, string num2) {
    if (num1.length() == num2.length()) {
        return num1 < num2;
    }
    return num1.length() < num2.length();
}

string sub(string num1, string num2) {
    string res = "";
    int i = num1.length() - 1, j = num2.length() - 1;
    int borrow = 0;
    while (i >= 0) {
        int x = num1[i] - '0';
        int y = j >= 0 ? num2[j] - '0' : 0;
        int cur = (x - y - borrow + 10) % 10;
        borrow = x - y - borrow < 0 ? 1 : 0;
        res += cur + '0';
        i--, j--;
    }
    reverse(res.begin(), res.end());
    // 删除前导0，res.size() - 1避免"000"情况
    int pos = 0;
    for (pos = 0; pos < res.size() - 1; ++pos) {
        if (res[pos] != '0') {
            break;
        }
    }
    return res.substr(pos);
}

int main(int argc, char* argv[]) {
    string num1, num2;
    cin >> num1 >> num2;
    string res = "";
    if (isLess(num1, num2)) {
        res.insert(0, "-");
        num1.swap(num2);
    }
    cout << res + sub(num1, num2) << "\n";
    return 0;
}
```

## 0727 双栈排序
给定一个乱序的栈，设计算法将其升序排列。  
ps: 允许额外使用一个栈来辅助操作 
```
输入
[4, 2, 1, 3]
输出
[1, 2, 3, 4]
```
```cpp
// 栈自底向上由小到大
stack<int> stackSort(stack<int>& s) {
    stack<int> res;
    while (!s.empty()) {
        int peak = s.top(); s.pop();
        while (!res.empty() && peak < res.top()) {
            int tmp = res.top(); res.pop();
            s.push(tmp);
        }
        res.push(peak);
    }
    return res;
}
```

## 0728 连续的n个数使其和最大
```cpp
int maxSum(vector<int>& nums, int n) {
    const int m = nums.size();
    if (n < 1 || n > m) {
        return -1;
    }
    int res = 0, i = 0;
    for (i = 0; i < n; ++i) {
        res += nums[i];
    }
    int curSum = res, left = 0, beginRes = 0;
    while (i < m) {
        curSum += nums[i] - nums[left];
        if (curSum > res) {
            res = curSum;
            beginRes = left + 1;  // 开始位置
        }
        i++;
        left++;
    }
    return res;
}
```

## 0729 阿拉伯数字转中文
```cpp
class Solution {
public:
    string num2cn(int n) {
        if (!n) {
            return "零";
        }
        bool isNeg = false;
        if (n < 0) {
            n = -n;
            isNeg = true;
        }
        string res = "";
        int pos = 0;
        bool needAddZero = false;
        while (n > 0) {
            int section = n % 10000;
            string sectionCn = section2cn(section);
            if (needAddZero) {
                res = "零" + res;
            }
            if (section == 0) {
                res = sectionCn + cnSectionChar[0] + res;
            } else {
                res = sectionCn + cnSectionChar[pos] + res;
            }
            needAddZero = (section > 0) && (section < 1000);
            n /= 10000;
            ++pos;
        }

        // 不可用: 中文字符不一定占2个字节
        // if (res.substr(0, 2).compare("一") == 0 && res.substr(2, 2).compare("十") == 0) {
        //     res = res.substr(2);
        // }
        
        if (res.find("一十") == 0) {
            res = res.substr(res.find("十"));
        }
        if (isNeg) {
            res.insert(0, "负");
        }
        return res;
    }

    string section2cn(int section) {
        string res = "";
        int pos = 0;
        bool addedZero = false;  // 每个小节最多只能出现1个"零"
        bool hasNonZeroBefore = false;
        while (section > 0) {
            int num = section % 10;
            if (num == 0) {
                // 从右向左: 如果有非零数字 且 还未添加"零"
                if (hasNonZeroBefore && !addedZero) {
                    res = cnChar[num] + res;
                    addedZero = true;
                }
            } else {
                hasNonZeroBefore = true;
                res = cnChar[num] + cnUnitChar[pos] + res;
            }
            section /= 10;
            ++pos;
        }
        return res;
    }

    vector<string> cnChar{"零", "一", "二", "三", "四", "五", "六", "七", "八", "九"};
    vector<string> cnUnitChar{"", "十", "百", "千"};
    vector<string> cnSectionChar{"", "万", "亿", "万亿", "亿亿"};
};
```
